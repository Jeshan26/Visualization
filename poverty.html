def classify_entity(name):
    if name == "World":
        return "World"
    if '(PIP)' in name or 'World' in name:
        return 'Region' if '(PIP)' in name else 'Global Aggregate'
    if '(urban)' in name:
        return 'Urban' 
    if '(rural)' in name:
        return 'Rural'
    else:
        return 'Country';
